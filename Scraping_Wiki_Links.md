``` r
Total.Links <- c()
for (i in links[7:32]){
  newLink <- str_remove(paste("https://en.wikipedia.org",i)," ")
  PageText <- read_html(newLink)
  PageLinks <- PageText%>%html_nodes('i > a')%>%html_attr('href')
  Total.Links <-  c(Total.Links,paste("https://en.wikipedia.org",PageLinks))
}
print(Total.Links)
```
